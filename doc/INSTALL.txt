Installation
============

Install Software
----------------

```sh
# xorg
pacman -S xautolock \
          xdg-utils \
          xlockmore \
          xorg-drivers \
          xorg-xkill \
          xorg-xmodmap \
          xterm

# audio
pacman -S alsa-firmware \
          alsa-utils \
          pulseaudio \
          pulseaudio-alsa

# terminal emulators
pacaur -S rxvt-unicode-pixbuf
pacman -S lxterminal \
          urxvt-perls \
          xfce4-terminal

# desktop
pacman -S adwaita-icon-theme \
          gnome-icon-theme \
          gnome-icon-theme-extras \
          gnome-menus \
          gnome-themes-extra \
          gtk-engines \
          lxappearance-obconf-gtk3 \
          lxinput-gtk3 \
          lxrandr-gtk3 \
          openbox \
          openbox-themes \
          materia-gtk-theme
pacaur -S obmenu-generator

# graphics (nvidia)
git clone https://gist.github.com/atweiden/701d14375996cf4823d369ef51169580

# desktop apps
pacman -S cbatticon \
          chromium \
          compton \
          conky \
          dmenu \
          feh \
          firefox \
          galculator \
          gmrun \
          gnumeric \
          gvim \
          keepassxc \
          leafpad \
          nitrogen \
          pcmanfm \
          redshift \
          tint2 \
          vlc \
          volumeicon \
          wmctrl

# fonts
pacaur -S monaco-powerline-font-git \
          ttf-google-fonts-git \
          ttf-monaco
```
